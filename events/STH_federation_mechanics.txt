	is_triggered_only = yes
	immediate = {
		every_country = {
			limit = {
				is_in_federation_with = root

			}
			set_country_flag = considering_federation
			country_event = { id = theFederation.1000 days = 14 random = 7 }
		}
		
		if = {
			limit = {
				any_country = {
					is_in_federation_with = root

						has_country_flag = wants_federation
