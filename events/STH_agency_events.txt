		name = STH_agency.1.a

		name = STH_agency.1.b
		hidden_effect = { set_timed_country_flag = { flag = recentIMECreateRequest days = 720 }  }

		name = STH_agency.5.a

		}

			create_fleet = { name = "Dummy" }

				set_owner = event_target:interspecies_medical_exchange

				set_leader = newLeader
