event = {
	id = STH_bajoran_story.11
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		# log = "STH_bajoran_story.11 started"
		random_country = {
			limit = { 
				has_country_flag = bajoran_republic
				OR = { is_country_type = default is_country_type = minorRace }
				NOT = { has_country_flag = bajoran_vedek_assembly_effect }
			}
			save_event_target_as = bajoran_republic
			country_event = { id = STH_bajoran_story.100 days = 15 }
		}
		ensure_bajoran_religion_country_exists = yes
		event_target:bajoranReligionCountry = {
			leader = { set_variable = { which = "vedekCount" value = 0 } }
			every_owned_leader = {
				limit = { has_leader_flag = bajoranVedek }
				prev.leader = { change_variable = { which = "vedekCount" value = 1 } }
			}
		}
	}
	after = {
		if = {
			limit = { exists = event_target:bajoran_republic event_target:bajoranReligionCountry = { leader = { check_variable = { which = "vedekCount" value < 5 } } } }

		}
		# log = "STH_bajoran_story.11 ended"	
	}
}

country_event = {
	id = STH_bajoran_story.4
	hide_window = yes
	is_triggered_only = yes
	trigger = { 
		has_country_flag = bajoranReligionCountry
		FROM = { 
			has_leader_flag = bajoranVedek 
			NOT = { has_leader_flag = bajoranKai }
		}
		any_country = { has_country_flag = bajoran_republic }
	}
	immediate = { 
		# log = "STH_bajoran_story.4 started"
		FROM = { save_event_target_as = deadBajoranVedek }
		random_country = {
			limit = { has_country_flag = bajoran_republic }
			save_event_target_as = bajoran_republic
		}
		leader = { set_variable = { which = "vedekCount" value = 0 } }
		every_owned_leader = {
			limit = { has_leader_flag = bajoranVedek }
			root = {
				if = {
					limit = { exists = leader }
					leader = { change_variable = { which = "vedekCount" value = 1 } }
				}
			}
		}
	}
	after = {
		if = {
			limit = { exists = leader leader = { check_variable = { which = "vedekCount" value < 8 } } }
			event_target:bajoran_republic = { country_event = { id = STH_bajoran_story.6 } }
			else = {
				event_target:bajoran_republic = { country_event = { id = STH_bajoran_story.7 } }
			}
		}
		# log = "STH_bajoran_story.4 ended"
	}
}
