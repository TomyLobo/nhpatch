generate_fallen_empire_start_pops = {
	if = {
		limit = { owner = { has_country_flag = init_spawned } }
		every_tile = {
			limit = { AND = { has_building = yes has_blocker = no has_grown_pop = no has_growing_pop = no } }
			create_pop = { species = owner_main_species }
		}
	}
}
